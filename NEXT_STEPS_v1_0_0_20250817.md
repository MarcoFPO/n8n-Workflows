```python
# ML-basierte Position Sizing
async def calculate_position_size(symbol: str, prediction: MLPrediction) -> float:
    base_size = portfolio_value * 0.02  # 2% base position
    
    # Confidence-based scaling
    confidence_multiplier = prediction.ensemble_confidence
    
    # Multi-horizon scoring
    horizon_score = (
        prediction.predictions["7d"].confidence * 0.4 +
        prediction.predictions["30d"].confidence * 0.3 +
        prediction.predictions["150d"].confidence * 0.2 +
        prediction.predictions["365d"].confidence * 0.1
    )
    
    return base_size * confidence_multiplier * horizon_score
```

```python
class MLTradingStrategy:
    def __init__(self, ml_service: MLAnalyticsService):
        self.ml_service = ml_service
        
    async def generate_signals(self, symbol: str) -> TradingSignal:
        predictions = await self.ml_service.get_predictions(symbol)
        
        # Multi-horizon signal generation
        short_term_signal = self.evaluate_short_term(predictions["7d"])
```
